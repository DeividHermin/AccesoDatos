1. Realizar una tabla html con el título "Situación de los libros" y en la que se muestren las columnas: código
de libro, título, año y situación (comprado o pendiente).
let $a := doc('/db/examenBDXML/libros.xml')/libreria/libro

return

<html>

    <head>
   
 </head>

    <body>

        <table border='1'>

	<caption>Situacion de los libros</caption>
            <tr>

                <td>Codigo</td>

                <td>Titulo</td>

                <td>Año</td>

                <td>Situacion</td>

            </tr>

            {for $b in $a

            return

            <tr>

                <td>{$b/data(@codigo)}</td>

                            then(<td> comprado </td>)

                }

            </tr>}

        </table>

    </body>

</html>

4. Realizar una tabla html con el título "Listado de libros" y que tenga las columnas: código, categoría, título,
autor, año y precio. En la columna autor deberán figurar el texto “varios" si hay más de un autor.

let $a := doc('/db/examenBDXML/libros.xml')/libreria/libro
return

<html>

    <head>
   
 </head>

    <body>

        <table border='1'>

	<caption>Listado de libros</caption>
            <tr>

                <td>Codigo</td>

                <td>Categoria</td>

                <td>Titulo</td>

                <td>Autor</td>

                <td>Año</td>

                <td>Precio</td>

            </tr>

            {for $b in $a

            return

            <tr>

                <td>{$b/data(@codigo)}</td>

                <td>{$b/data(@categoria)}</td>

                <td>{$b/titulo/text()}</td>

                <td>{if(count($b/autor)>1)

                    then('Varios')

                    else($b/autor/text())

                }</td>

                <td>{$b/anyo/text()}</td>

                <td>{$b/precio/text()}</td>

            </tr>}

        </table>

    </body>

</html>
